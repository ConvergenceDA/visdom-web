API playground

Open your dev tools on this page and you can play with the JavaScript API. Check out the wiki for more info on query URLs, or start with these examples:

api.get("query?/s/basics|kw.mean&therm.mean/f/zip5|=93304", function(err, data) {
  console.log(data);
});
api.query({
  columns: ["kw.mean", "therm.mean"],
  filter: {
    zip5: "=93304"
  }
}, function(err, data) {
  console.log(data);
});
api.query({
  columns: ["kw.mean", "therm.mean"],
  agg: {
    zip5: "mean"
  },
  cum: true
}, function(err, data) {
  console.log(data);
});
